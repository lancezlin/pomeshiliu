DROP TABLE IF EXISTS product;
CREATE TABLE product (
  prod_id BIGINT NOT NULL,
  parent_id INT NULL,
  name VARCHAR(255) NULL,
  msrp DOUBLE NULL,
  price DOUBLE NULL,
  cat_path VARCHAR(1000) NULL,
  short_desc TEXT NULL,
  long_desc TEXT NULL,
  brand VARCHAR(255) NULL,
  thum_img VARCHAR(1000) NULL,
  med_img VARCHAR(1000) NULL,
  large_img VARCHAR(1000) NULL,
  track_url VARCHAR(1000) NULL,
  dolar_ship TINYINT(1) NULL,
  std_ship_rate DOUBLE NULL,
  two_day_ship_rate DOUBLE NULL,
  over_night_ship_rate DOUBLE NULL,
  market_place TINYINT(1) NULL,
  ship_to_store TINYINT(1) NULL,
  free_ship_to_store TINYINT(1) NULL,
  model_num VARCHAR(255) NULL,
  url VARCHAR(1000) NULL,
  customer_rating VARCHAR(20) NULL,
  num_of_review INT NULL,
  rating_img VARCHAR(255) NULL,
  cat_node VARCHAR(255) NULL,
  roll_back TINYINT(1) NULL,
  bundle TINYINT(1) NULL,
  clearance TINYINT(1) NULL,
  pre_order TINYINT(1) NULL,
  stock VARCHAR(20) NULL,
  avail_online TINYINT(1) NULL,
  PRIMARY KEY (prod_id));
